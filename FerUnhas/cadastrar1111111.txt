function cadastrarUsuario() {
    const fname = document.getElementById('firstname').value;
    const lname = document.getElementById('lastname').value;
    const email = document.getElementById('email').value;
    const senha = document.getElementById('senha').value;
    const number = document.getElementById('number').value;
    const cep = document.getElementById('cep').value;
    const rua = document.getElementById('rua').value;
    const bairro = document.getElementById('bairro').value;
    const num = document.getElementById('num').value;
    const cidade = document.getElementById('cidade').value;

    const UsuariorequestBody = {
        fname,
        lname,
        email,
        senha,
        number
    };

    const EnderecorequestBody = {
        cep,
        rua,
        bairro,
        num,
        cidade
    };

    fetch('http://localhost:8080/usuario', {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json',
        },
        body: JSON.stringify(UsuariorequestBody),
    })
    fetch('http://localhost:8080/endereco', {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json',
        },
        body: JSON.stringify(EnderecorequestBody),
    })
        .then(response => response.json())
        .then(data => {
            alert('Usu치rio cadastrado com sucesso!');
            // Redefine cada campo manualmente
            document.getElementById('firstname').value = '';
            document.getElementById('lastname').value = '';
            document.getElementById('email').value = '';
            document.getElementById('senha').value = '';
            document.getElementById('number').value = '';
            document.getElementById('cep').value = '';
            document.getElementById('rua').value = '';
            document.getElementById('bairro').value = '';
            document.getElementById('num').value = '';
            document.getElementById('cidade').value = '';
            
            // Recarrega a p치gina, se necess치rio
            location.reload();
        })
        .catch(error => {
            console.error('Erro ao cadastrar usu치rio:', error);
        });
}
